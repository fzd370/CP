int distinct_chars(const char*c){
    char ch[26];
    int i,count=0;
    for(i=0;i<26;i++)
    ch[i]=0;
    for(i=0;c[i]!='\0';i++){
        ch[c[i]-'a']++;
    }
    for(i=0;i<26;i++)
    if(ch[i]!=0)
    count++;

    return count;
}

int sort_by_number_of_distinct_characters(const char* a, const char* b) {
    int len=distinct_chars(a)-distinct_chars(b);
    if(len>0)
    return 1;
    else if(len==0){
    if(strcmp(a,b)>0)
    return  1;
    else return 0;
    }
    else
    return 0;
}

void string_sort(char** arr,const int len,int (*cmp_func)(const char* a, const char* b)){
    int i,j;
    for(i=0;i<len;i++){
        for(j=0;j<len-1;j++){
            if(cmp_func(arr[j],arr[j+1])){
                char *temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
    

}
